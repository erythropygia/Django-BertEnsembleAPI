                      String url = "http://localhost:8000/process";
                      String inputText = "Send text here.";
                      String auth_token = "YOUR_TOKEN_HERE";
                      StringBuilder builder = new StringBuilder();
                          
                      for (int i = 0; i < inputText.length(); i++) {
                        
                          if (inputText.charAt(i) == ' ') {
                            
                                      builder.append("%20");
                                      
                                  } else {
                                    
                                      builder.append(inputText.charAt(i));
                                    
                                  }
                              }
                              
                              String convertedString = builder.toString();
                          

                      try {
                          URL urlObj = new URL(url + "?text=" + convertedString);
                          HttpURLConnection conn = (HttpURLConnection) urlObj.openConnection();
                          conn.setRequestMethod("GET");
                          conn.setRequestProperty("Authorization", "Token " + auth_token);

                          if (conn.getResponseCode() == HttpURLConnection.HTTP_OK) {
                              BufferedReader in = new BufferedReader(new InputStreamReader(conn.getInputStream()));
                              String response = in.readLine();
                              String result = response.split("\"result\":")[1].split("}")[0];
                              System.out.println("Result: " + result);
                              in.close();
                          } else {
                              System.out.println("Error: " + conn.getResponseMessage());
                          }
                              conn.disconnect();
                          } catch (Exception e) {
                              System.out.println("Error: " + e.getMessage());
                          }

  • JavaScript 

    
                        let url = 'http://localhost:8000/process';
                        let inputText = 'Send text here.';
                        let auth_token = 'YOUR_TOKEN_HERE';
                        
                        let xhr = new XMLHttpRequest();
                        xhr.open("GET", `${url}?text=${inputText}`);
                        xhr.setRequestHeader("Authorization", `Token ${auth_token}`);
                        xhr.onreadystatechange = function () {
                          if (xhr.readyState === 4) {
                            if (xhr.status === 200) {
                                  let response = JSON.parse(xhr.responseText);
                                  let result = response.result;
                                  document.getElementById("result").innerHTML = "Result: " + result;
                            } else {
                                  console.log("Error:", xhr.statusText);
                                  document.getElementById("result").innerHTML = "Error: " + xhr.statusText;
                            }
                          }
                        };
                        xhr.send();
